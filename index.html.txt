// Graphique camembert SVG simple
function PieChart({ data, size=120 }) {
  let cumul = 0;
  const total = data.reduce((a,d)=>a+d.value,0);
  const slices = data.map(d => {
    const start = cumul / total * 360;
    const angle = d.value / total * 360;
    cumul += d.value;
    return { ...d, start, angle };
  });

  function slice(start, angle, r=46, cx=60, cy=60) {
    const s = (start - 90) * Math.PI / 180;
    const e = (start + angle - 90) * Math.PI / 180;
    const x1 = cx + r * Math.cos(s), y1 = cy + r * Math.sin(s);
    const x2 = cx + r * Math.cos(e), y2 = cy + r * Math.sin(e);
    const large = angle > 180 ? 1 : 0;
    return `M${cx},${cy} L${x1},${y1} A${r},${r} 0 ${large},1 ${x2},${y2} Z`;
  }

  return (
    <svg width={size} height={size} viewBox="0 0 120 120">
      <circle cx="60" cy="60" r="46" fill={C.border}/>
      {slices.map((s,i) => (
        <path key={i} d={slice(s.start, s.angle)} fill={s.color} opacity="0.9"/>
      ))}
      <circle cx="60" cy="60" r="28" fill={C.surface}/>
    </svg>
  );
}

  const total = base + extra;
  const ttc   = (total*1.2).toFixed(2).replace(".",",");

  const STEPS = ["Trajet & véhicule","Prestations","Récapitulatif","Confirmation"];

  const next = () => {
    if(step===1 && (!depart||!arrivee||!date)) {
      showToast("Veuillez renseigner le trajet et la date.","error"); return;
    }
    if(step===3) showToast("Mission envoyée avec succès !","success");
    setStep(s=>Math.min(s+1,4));
  };

  return (
    <div style={{maxWidth:780,margin:"0 auto",animation:"fadeUp 0.3s ease"}}>

      {/* Header */}
      <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:28}}>
        <div>
          <h1 style={{fontFamily:"Syne,sans-serif",fontSize:26,fontWeight:700,color:C.text,margin:0}}>Nouvelle mission</h1>
          <p style={{color:C.textMuted,fontSize:13,marginTop:4}}>Renseignez les informations pour générer votre devis</p>
        </div>
        <Btn v="ghost" onClick={onBack}>Annuler</Btn>
      </div>

      {/* Stepper */}
      <div style={{display:"flex",alignItems:"center",marginBottom:28}}>
        {STEPS.map((label,i) => {
          const n=i+1, done=n<step, active=n===step;
          return (
            <div key={n} style={{display:"flex",alignItems:"center",flex:n<STEPS.length?1:0}}>
              <div style={{display:"flex",alignItems:"center",gap:8}}>
                <div style={{width:28,height:28,borderRadius:"50%",background:done?C.green:active?C.primary:"transparent",border:`2px solid ${done?C.green:active?C.primary:C.border}`,color:done||active?"#fff":C.textMuted,display:"flex",alignItems:"center",justifyContent:"center",fontSize:11,fontWeight:700,flexShrink:0,transition:"all 0.3s"}}>
                  {done?"✓":n}
                </div>
                <span style={{fontSize:12,fontWeight:active?600:400,color:active?C.primary:done?C.green:C.textMuted,whiteSpace:"nowrap"}}>{label}</span>
              </div>
              {n<STEPS.length && <div style={{flex:1,height:2,background:done?C.green:C.border,margin:"0 10px",transition:"background 0.4s"}}/>}
            </div>
          );
        })}
      </div>

      {/* ÉTAPE 1 */}
      {step===1 && (
        <Card>
          <div style={{padding:"20px 24px",borderBottom:`1px solid ${C.border}`}}>
            <span style={{fontSize:11,fontWeight:700,color:C.primary,textTransform:"uppercase",letterSpacing:"0.1em"}}>Étape 1 — Trajet & véhicule</span>
          </div>
          <div style={{padding:24,display:"flex",flexDirection:"column",gap:16}}>
            <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:16}}>
              <Input label="Adresse de départ" value={depart} onChange={setDepart} placeholder="Ex : 1 rue de Lyon, 51100 Reims"/>
              <Input label="Adresse d'arrivée" value={arrivee} onChange={setArrivee} placeholder="Ex : 14 avenue Foch, 13008 Marseille"/>
            </div>
            <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:16}}>
              <Input label="Date de départ souhaitée" type="date" value={date} onChange={setDate}/>
              <Input label="Immatriculation" value={immat} onChange={v=>setImmat(v.toUpperCase())} placeholder="Ex : AB123CD" note="Les informations du véhicule seront récupérées automatiquement"/>
            </div>
            {immat.length >= 7 && (
              <div style={{background:C.primaryLight,border:`1px solid ${C.accent}22`,borderRadius:8,padding:"12px 16px",display:"flex",gap:16,alignItems:"center"}}>
                <div style={{fontSize:12,color:C.textSec}}>Véhicule détecté :</div>
                <Input label="" value={marque} onChange={setMarque} placeholder="Marque et modèle"/>
                <div style={{fontSize:11,color:C.textMuted,whiteSpace:"nowrap"}}>Modifiable si incorrect</div>
              </div>
            )}
            {depart && arrivee && (
              <div style={{display:"inline-flex",alignItems:"center",gap:10,background:C.primaryLight,border:`1px solid ${C.accent}33`,borderRadius:8,padding:"10px 16px"}}>
                <span style={{fontSize:13,color:C.textSec}}>Distance estimée :</span>
                <span style={{fontFamily:"Syne,sans-serif",fontSize:18,fontWeight:700,color:C.primary}}>412 km</span>
                <span style={{fontSize:12,color:C.textMuted}}>— Tranche tarifaire : 400–500 km</span>
              </div>
            )}
          </div>
          <div style={{padding:"14px 24px",borderTop:`1px solid ${C.border}`,display:"flex",justifyContent:"flex-end"}}>
            <Btn onClick={next}>Suivant —</Btn>
          </div>
        </Card>
      )}

      {/* ÉTAPE 2 */}
      {step===2 && (
        <Card>
          <div style={{padding:"20px 24px",borderBottom:`1px solid ${C.border}`}}>
            <span style={{fontSize:11,fontWeight:700,color:C.primary,textTransform:"uppercase",letterSpacing:"0.1em"}}>Étape 2 — Prestations souhaitées</span>
          </div>
          <div style={{padding:24}}>
            <div style={{display:"grid",gridTemplateColumns:"repeat(3,1fr)",gap:10,marginBottom:8}}>
              {PRESTS.map(p => {
                const sel = prests[p.key];
                return (
                  <div key={p.key} onClick={()=>toggle(p.key)}
                    style={{border:`1.5px solid ${sel?C.primary:C.border}`,background:sel?C.primaryLight:C.surface,borderRadius:10,padding:"14px 16px",cursor:"pointer",transition:"all 0.15s",position:"relative",userSelect:"none"}}>
                    {sel && (
                      <div style={{position:"absolute",top:10,right:10,width:18,height:18,background:C.primary,borderRadius:"50%",display:"flex",alignItems:"center",justifyContent:"center",color:"#fff",fontSize:10,fontWeight:700}}>✓</div>
                    )}
                    <div style={{fontSize:13,fontWeight:600,color:sel?C.primary:C.text,marginBottom:4}}>{p.label}</div>
                    <div style={{fontSize:12,fontWeight:700,color:p.inclus?C.textMuted:C.accent}}>{p.inclus?"Inclus":`+${p.prix} €`}</div>
                  </div>
                );
              })}
            </div>
          </div>
          <div style={{padding:"14px 24px",borderTop:`1px solid ${C.border}`,display:"flex",justifyContent:"space-between"}}>
            <Btn v="ghost" onClick={()=>setStep(1)}>— Retour</Btn>
            <Btn onClick={next}>Suivant —</Btn>
          </div>
        </Card>
      )}

      {/* ÉTAPE 3 */}
      {step===3 && (
        <Card>
          <div style={{padding:"20px 24px",borderBottom:`1px solid ${C.border}`}}>
            <span style={{fontSize:11,fontWeight:700,color:C.primary,textTransform:"uppercase",letterSpacing:"0.1em"}}>Étape 3 — Récapitulatif & devis</span>
          </div>
          <div style={{padding:24}}>
            <div style={{background:C.bg,borderRadius:10,padding:16,marginBottom:20}}>
              <div style={{fontFamily:"Syne,sans-serif",fontSize:18,fontWeight:700,color:C.text}}>{depart||"—"} → {arrivee||"—"} · 412 km</div>
              <div style={{fontSize:13,color:C.textSec,marginTop:4}}>{date||"—"} · Hexagone Confluence</div>
            </div>
            <div style={{border:`1px solid ${C.border}`,borderRadius:10,overflow:"hidden",marginBottom:16}}>
              <div style={{display:"flex",justifyContent:"space-between",padding:"12px 16px",borderBottom:`1px solid ${C.border}`,fontSize:13,color:C.textSec}}>
                <span>Convoyage (412 km)</span><strong style={{color:C.text}}>{base} € HT</strong>
              </div>
              {PRESTS.filter(p=>prests[p.key]).map(p=>(
                <div key={p.key} style={{display:"flex",justifyContent:"space-between",padding:"12px 16px",borderBottom:`1px solid ${C.border}`,fontSize:13,color:C.textSec}}>
                  <span>{p.label}</span><strong style={{color:p.inclus?C.green:C.text}}>{p.inclus?"Inclus":`+${p.prix} € HT`}</strong>
                </div>
              ))}
              <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",padding:"14px 16px"}}>
                <div>
                  <div style={{fontSize:11,fontWeight:700,color:C.textMuted,textTransform:"uppercase",letterSpacing:"0.08em"}}>Total estimé</div>
                  <div style={{fontSize:11,color:C.textMuted,marginTop:2}}>Soit {ttc} € TTC</div>
                </div>
                <div style={{fontFamily:"Syne,sans-serif",fontSize:30,fontWeight:700,color:C.primary}}>{total} € HT</div>
              </div>
            </div>
            <div style={{background:"#FFFBEB",border:"1px solid #F6E05E",borderRadius:8,padding:"12px 14px",fontSize:13,color:"#744210"}}>
              <strong>Conditions d'annulation :</strong> Plus de 48h : gratuite · Entre 24h et 48h : 30% · Jour J : 60%
            </div>
          </div>
          <div style={{padding:"14px 24px",borderTop:`1px solid ${C.border}`,display:"flex",justifyContent:"space-between"}}>
            <Btn v="ghost" onClick={()=>setStep(2)}>— Retour</Btn>
            <Btn onClick={next}>Valider et envoyer</Btn>
          </div>
        </Card>
      )}

      {/* ÉTAPE 4 */}
      {step===4 && (
        <Card>
          <div style={{padding:"48px 28px",textAlign:"center"}}>
            <div style={{width:56,height:56,borderRadius:"50%",background:C.greenBg,border:`2px solid ${C.green}`,display:"flex",alignItems:"center",justifyContent:"center",margin:"0 auto 20px",fontSize:22,color:C.green,fontWeight:700}}>✓</div>
            <h2 style={{fontFamily:"Syne,sans-serif",fontSize:24,fontWeight:700,color:C.text,marginBottom:10}}>Mission envoyée</h2>
            <p style={{fontSize:14,color:C.textSec,maxWidth:400,margin:"0 auto 24px"}}>
              Votre demande a été transmise à Hexagone Confluence. Vous recevrez une confirmation dès qu'un convoyeur sera assigné.
            </p>
            <div style={{display:"inline-flex",alignItems:"center",gap:10,background:C.bg,border:`1px solid ${C.border}`,borderRadius:8,padding:"10px 20px",marginBottom:28}}>
              <span style={{fontSize:12,color:C.textMuted}}>Référence</span>
              <strong style={{fontFamily:"Syne,sans-serif",fontSize:16,color:C.primary}}>#HC-2026-{ref}</strong>
            </div>
            <div style={{display:"flex",gap:10,justifyContent:"center",flexWrap:"wrap"}}>
              <Btn v="outline" onClick={()=>showToast("Génération PDF — à brancher côté développeur","success")}>Télécharger le bon de commande</Btn>
              <Btn onClick={onBack}>Retour au tableau de bord</Btn>
            </div>
          </div>
        </Card>
      )}
    </div>
  );
}
